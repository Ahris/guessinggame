CREATE DATABASE rps_data;

CREATE TABLE singlePlayer (
    pId MEDIUMINT UNSIGNED NOT NULL AUTO_INCREMENT,
	userId MEDIUMINT UNSIGNED NOT NULL,
    turnNum MEDIUMINT UNSIGNED NOT NULL,
    timeElapsed DECIMAL(10, 3) NOT NULL,
    playerChoice ENUM('rock', 'paper', 'scissors') NOT NULL,
    cpuChoice ENUM('rock', 'paper', 'scissors') NOT NULL,
    turnScore DECIMAL(10, 3) NOT NULL,
    totalScore DECIMAL(10, 3) NOT NULL,
    result ENUM('win', 'tie', 'lose') NOT NULL,
    totalGames MEDIUMINT UNSIGNED NOT NULL,
    scoreMode ENUM('standard', 'stochastic') NOT NULL,
    oppMode ENUM('view', 'hide') NOT NULL,
    PRIMARY KEY (pId)
) engine = INNODB DEFAULT character SET = utf8 COLLATE = utf8_general_ci;
